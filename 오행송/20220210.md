T = int(input())
def move(aaa):
    temp = aaa[len(aaa)-1]
    for i in range(len(aaa)-1,0,-1):
        aaa[i] = aaa[i - 1]
    aaa[0] = temp
    return aaa

for i in range(0,T):
    N = int(input())
    SC = list(input().split())


    max1=0
    for j in range(7):
        move(SC)
        SCC = ''
        for j in SC:
            SCC = SCC + j
        if int(SCC) > int(max1):
            max1=SCC

    NN=0
    day=0

    while NN < N:
        for j in range(0,7):
            day += 1
            if max1[j] == '1':
                NN += 1

            if NN == N:
                break
    print('#{} {}'.format(i+1,day))
